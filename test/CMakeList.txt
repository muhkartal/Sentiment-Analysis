cmake_minimum_required(VERSION 3.10)

# Find GTest package
find_package(GTest REQUIRED)
include_directories(${GTEST_INCLUDE_DIRS})

# Add the test executable
add_executable(sentiment_tests
    test_preprocessor.cpp
    test_data_loader.cpp
    test_feature_extractor.cpp
    test_naive_bayes.cpp
)

# Link against Google Test and project libraries
target_link_libraries(sentiment_tests
    ${GTEST_LIBRARIES}
    pthread
)

# Include the project's include directory
target_include_directories(sentiment_tests PRIVATE
    ${CMAKE_SOURCE_DIR}/include
)

# Add test to be run with CTest
add_test(NAME sentiment_tests COMMAND sentiment_tests)
